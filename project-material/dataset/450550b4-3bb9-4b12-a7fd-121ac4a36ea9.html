- Find the information gain on the first split with **Entropy** according to the formula from the lecture notes (given below) Information Gain = entropy(parent) - [average entropy(children)]
ChatGPT
python
from math import log2

# Calculate entropy
def calculate_entropy(labels):
    class_counts = labels.value_counts()
    total_samples = len(labels)
    entropy_value = 0

    for count in class_counts:
        probability = count / total_samples
        entropy_value -= probability * log2(probability)

    return entropy_value

# Calculate information gain on the first split
parent_entropy = calculate_entropy(y_train)

# Get the first split features and values
feature = X_train.columns[best_dt_classifier.tree_.feature[0]]
split_value = best_dt_classifier.tree_.threshold[0]

# Create left and right child datasets based on the first split
left_child = X_train[X_train[feature] <= split_value]
right_child = X_train[X_train[feature] > split_value]

# Calculate entropy for left and right children
left_child_entropy = calculate_entropy(y_train.loc[left_child.index])
right_child_entropy = calculate_entropy(y_train.loc[right_child.index])

# Calculate average entropy of children
average_child_entropy = (len(left_child) / len(X_train)) * left_child_entropy + (len(right_child) / len(X_train)) * right_child_entropy

# Calculate information gain
information_gain = parent_entropy - average_child_entropy

print("Information Gain on the first split (Entropy):", information_gain)

To calculate the correlations of features with the target variable 'health_metrics' and plot the results in a heatmap, you can use the following Python code:

python
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Load the dataset
csv_file_path = "/content/cs412_hw1_dataset.csv"
df = pd.read_csv(csv_file_path)

# Mapping dictionaries
sex_map = {'female': 1, 'male': 0}
island_map = {'Biscoe': 1, 'Dream': 2, 'Torgensen': 3}
diet_map = {'fish': 1, 'krill': 2, 'squid': 3, "parental": 4}
life_stage_map = {'chick': 1, 'juvenile': 2, 'adult': 3}
health_metrics_map = {'healthy': 1, 'overweight': 2, 'underweight': 3}
species_map = {'Adelie': 1, 'Chinstrap': 2, 'Gentoo': 3}

# Fill missing values with the most common value in the corresponding column
for column in df.columns:
    if df[column].dtype == 'object':
        most_common_value = df[column].mode()[0]
        df[column].fillna(most_common_value, inplace=True)

# Applying the mappings to the categorical columns
df['sex'] = df['sex'].map(sex_map)
df['island'] = df['island'].map(island_map)
df['diet'] = df['diet'].map(diet_map)
df['life_stage'] = df['life_stage'].map(life_stage_map)
df['health_metrics'] = df['health_metrics'].map(health_metrics_map)
df['species'] = df['species'].map(species_map)

# Calculate correlations between features and the target variable
correlations = df.corr()['health_metrics'].sort_values(ascending=False)

# Plot correlations in a heatmap
plt.figure(figsize=(12, 8))
sns.heatmap(df.corr(), annot=True, cmap='coolwarm', linewidths=0.5)
plt.title('Correlation Heatmap')
plt.show()

# Display the correlations
print("Correlations with 'health_metrics':")
print(correlations)

This code will calculate the correlations of all features with the target variable 'health_metrics' and display them in a heatmap. Strong positive or negative correlations with the target variable will be highlighted in the heatmap, helping you identify which features are most relevant for predicting penguin health conditions.

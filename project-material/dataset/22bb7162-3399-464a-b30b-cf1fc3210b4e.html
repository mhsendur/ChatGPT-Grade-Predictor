from sklearn.tree import _tree
import numpy as np

def calculate_entropy(labels):
    class_counts = np.bincount(labels)
    probabilities = class_counts / len(labels)
    entropy_value = -np.sum([p * np.log2(p) if p > 0 else 0 for p in probabilities])
    return entropy_value

def calculate_information_gain(X, y, feature_index, threshold):
    parent_entropy = calculate_entropy(y)

    # Split the data based on the given feature and threshold
    left_child_mask = X[:, feature_index] <= threshold
    right_child_mask = ~left_child_mask

    left_child_entropy = calculate_entropy(y[left_child_mask])
    right_child_entropy = calculate_entropy(y[right_child_mask])

    # Calculate the average entropy of the children
    average_child_entropy = (np.sum(left_child_mask) / len(y)) * left_child_entropy + \
                            (np.sum(right_child_mask) / len(y)) * right_child_entropy

    # Calculate information gain
    information_gain = parent_entropy - average_child_entropy
    return information_gain
